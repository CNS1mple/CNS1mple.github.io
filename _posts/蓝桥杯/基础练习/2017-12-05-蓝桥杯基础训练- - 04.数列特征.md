问题描述  
给出n个数，找出这n个数的最大值，最小值，和。  

输入格式  
第一行为整数n，表示数的个数。  

第二行有n个数，为给定的n个数，每个数的绝对值都小于10000。  

输出格式  
输出三行，每行一个整数。第一行表示这些数中的最大值，第二行表示这些数中的最小值，第三行表示这些数的和。  
样例输入  
5  
1 3 -2 4 5  

样例输出  
5  
-2  
11  

数据规模与约定  
1 <= n <= 10000。  
   

  
  
  
当时的C语言代码很丑陋，见谅...XD按要求输出即可...

#include <stdio.h>

void swap(int A[], int i, int j){
	int temp;
	temp=A[i];
	A[i]=A[j];
	A[j]=temp;
}

void maopao(int A[], int n){
	int i, j;
	for(i=0;i<n;i++){
		for(j=0;j<n-i-1;j++){
			if(A[j]>A[j+1]){
				swap(A,j,j+1);
			}
		}
	}
}

int main(int argc, char *argv[]) {
	int n;
	int A[10005];
	int i,j;
	long long sum=0;
	scanf("%d\n",&n);
	for(i=0;i<n;i++){
		scanf("%d",&A[i]);
		sum+=A[i];
	}	
		maopao(A,n);
		printf("%d\n",A[n-1]);
		printf("%d\n",A[0]);
		printf("%lld",sum);
		
		return 0;

}
